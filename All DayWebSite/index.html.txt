<script>
    const BASE_URL = "https://vpic.nhtsa.dot.gov/api/vehicles";
    const yearSelect = document.getElementById('year');
    const makeSelect = document.getElementById('make');
    const modelSelect = document.getElementById('model');

    for (let i = new Date().getFullYear() + 1; i >= 1990; i--) {
        yearSelect.add(new Option(i, i));
    }

    async function loadMakes() {
        try {
            const response = await fetch(`${BASE_URL}/GetAllMakes?format=json`);
            const data = await response.json();
            const sorted = data.Results.sort((a, b) => a.Make_Name.localeCompare(b.Make_Name));
            sorted.forEach(m => makeSelect.add(new Option(m.Make_Name, m.Make_Name)));
        } catch (e) { console.error("Error loading makes"); }
    }
    loadMakes();

    makeSelect.addEventListener('change', updateModels);
    yearSelect.addEventListener('change', updateModels);

    async function updateModels() {
        const year = yearSelect.value;
        const make = makeSelect.value;
        if (!year || !make) return;

        modelSelect.innerHTML = '<option>Loading...</option>';
        modelSelect.disabled = true;

        try {
            const response = await fetch(`${BASE_URL}/GetModelsForMakeYear/make/${make}/modelyear/${year}?format=json`);
            const data = await response.json();
            modelSelect.innerHTML = '<option value="">Select Model</option>';
            modelSelect.disabled = false;
            data.Results.forEach(m => modelSelect.add(new Option(m.Model_Name, m.Model_Name)));
        } catch (e) { modelSelect.innerHTML = '<option>Error loading models</option>'; }
    }
</script>
